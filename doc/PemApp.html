post '/deploy_mod' do
  content_type 'application/json'
  data = JSON.parse(request.body.read)

  begin
    data.each do |m, v|
      pem.deploy_mod(m, v)
    end
    { 'status' => 'successful' }.to_json
  rescue
    { 'status' => 'failed' }.to_json
  end
end

post '/envs/:name/create' do
  content_type 'application/json'
  data = JSON.parse(request.body.read)
  e = PemEnv.new(params[:name], pem)

  begin
    e.deploy(data)
    { 'status' => 'successful' }.to_json
  rescue
    { 'status' => 'failed' }.to_json
  end
end
